import java.util.*;

public class FIFOPageReplacement 
{
    public static void main(String[] args) 
    {
        Scanner sc = new Scanner(System.in);

        System.out.print("Enter number of frames: ");
        int framesCount = sc.nextInt();

        System.out.print("Enter number of pages: ");
        int pagesCount = sc.nextInt();

        int[] pages = new int[pagesCount];
        System.out.println("Enter the reference string (page numbers):");
        
	for (int i = 0; i < pagesCount; i++) 
	{
            pages[i] = sc.nextInt();
        }

        // Queue to represent pages in memory
        Queue<Integer> queue = new LinkedList<>();
        HashSet<Integer> memory = new HashSet<>();

        int pageFaults = 0;

        System.out.println("\nPage\tFrames");
        System.out.println("--------------------");

        for (int i = 0; i < pagesCount; i++) 
	{
            int page = pages[i];

            // If page is not present in memory (page fault)
            if (!memory.contains(page)) 
	    {
                // If memory is full, remove the oldest page
                if (memory.size() == framesCount) 
		{
                    int removed = queue.poll();
                    memory.remove(removed);
                }

                // Add new page
                queue.add(page);
                memory.add(page);
                pageFaults++;
            }

            // Print current frame content
            System.out.print(page + "\t");
            printQueue(queue, framesCount);
        }

        System.out.println("\nTotal Page Faults: " + pageFaults);
    }

    // Helper method to print frames
    static void printQueue(Queue<Integer> queue, int frameCount) 
	{
        List<Integer> list = new ArrayList<>(queue);
        for (int i = 0; i < frameCount; i++) 
		{
            if (i < list.size())
                System.out.print(list.get(i) + " ");
            else
                System.out.print("- ");
        }
        System.out.println();
    }
}
